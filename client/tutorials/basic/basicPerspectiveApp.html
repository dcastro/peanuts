Basic Perspective App Setup

        
            // Find the div on the page you would like the application to appear in.
            var appElement = document.getElementById("app");

            // Configure your application, here we are asking for a WebGL window
            // with a perspective view, by default Peanuts will create a default
            // scene with some shapes and some three orbit controls so you know 
            // everythign is working.
            var app = new Peanuts.Three.App.Builder(appElement)
                     .withGLRenderer()
                     .withPerspectiveView()
                     .build();

            // This tells the engine to start running and render your stuff.
            // In this case it's just to draw a few shapes :P but it gets more 
            // exciting, I promise :)
            app.start();
